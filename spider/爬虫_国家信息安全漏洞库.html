    网址：国家信息安全漏洞库 (cnnvd.org.cn) (opens new window)

    使用的技术

    selenium
xpath

requests库
lxml 库
xlrd 库
xlutils 库

    # 说明

    先将工作簿1.xlsx的数据读取出来，在根据这些数据 输入漏洞名称或漏洞编号点击搜索，将展示的数据（标题和CNNVD编号）爬取下来

    注意：用漏洞名称爬取的时候，数据可能不止一个或一页

    # 思路

    # 读取xlsx中的数据

    把表格中的数据读取出来，有的是漏洞名称，有的是漏洞CNNVD编号，并且有的单元格不止一个CNNVD编号需要切分出来。

    # 根据漏洞名称 使用 selenium 爬取

    用漏洞名称搜索出来会有较多数据，使用 selenium 主要是为了反爬，没有使用无头浏览器。爬取出来的标题可能过长，页面显示不完全，使用...省略了，查看源代码，可以在a标签中的onmouseover中看到完整的标题

    # 根据漏洞编号爬取

    每一个编号对应一个漏洞，使用requests库+xpath解析

    # 保存数据到xls

    将爬取的标题和CNNVD编号写入到表格中

import requests
from lxml import etree
from selenium import webdriver
from selenium.webdriver.common.by import By
import time
import xlrd
from lxml.html import fromstring, tostring

from xlutils.copy import copy

name_values = []  # 漏洞名称集合
no_values = []  # 漏洞编号集合


# 创建 Chrome（不放在全局变量会闪退）
chrome = webdriver.Chrome()
chrome.get('http://www.cnnvd.org.cn/web/vulnerability/queryLds.tag')

# 1.读取xlsx中的数据
def read_data(path):
    data = xlrd.open_workbook(path)
    table = data.sheets()[0]  # 通过索引顺序获取
    # 获取A列所有数据
    values = table.col_values(0, start_rowx=0, end_rowx=None)  # 返回由该列中所有单元格的数据组成的列表

    for value in values:
        value = str(value).strip()
        # 查找编号
        if value.find('CNNVD') != -1:
            # 有的里面是用换行符分开的
            if value.find('\n') != -1:
                temps = value.split('\n')
                for temp in temps:
                    no_values.append(temp)
            # 有的里面使用空格分开的
            elif value.find(' ') != -1:
                temps = value.split(' ')
                for temp in temps:
                    no_values.append(temp)
            # 只有单独的一个
            else:
                no_values.append(value)
        # 查找名称
        else:
            # 去重
            if value not in name_values and value != '':
                name_values.append(value)

    return name_values,no_values

# 2.根据漏洞名称 使用 selenium 爬取
def spider_by_selenium(name):
    """
    url: http://www.cnnvd.org.cn/web/vulnerability/queryLds.tag
    """

    # 存放标题
    new_titles = []
    # 存放CNNVD
    new_nos = []


    # 获取漏洞名称输入框，默认用id寻找
    name_input = chrome.find_element(value='qcvCname')
    # 获取搜索按钮，用class
    search_button = chrome.find_element(by=By.CLASS_NAME,value='bd_b')

    # 先清空搜索框数据
    name_input.clear()

    # 向搜索框输入数据
    name_input.send_keys(name)
    time.sleep(1)

    # 点击提交按钮
    search_button.click()
    # 记录页码
    page_num = 1
    while True:
        print('========== 正在爬取名称为 "{}" 第{}页 的标题和编号 =========='.format(name, page_num))

        # 爬取标题和CNNVD编号
        # 当前页面源码
        page_source = chrome.page_source
        tree = etree.HTML(page_source)

        # 两个 xpath
        name_xpath = '//div[@class="list_list"]//li//div[@class="fl"]/a/text()'
        no_xpath = '//div[@class="list_list"]//li//div[@class="fl"]/p/a[1]/text()'

        # CNNVD编号
        nos = tree.xpath(no_xpath)
        for no in nos:
            new_nos.append(no)


        # 处理标题
        temp_list = deal_title_by_selenium(page_source)
        for temp in temp_list:
            new_titles.append(temp)


        # preceding-sibling（当前节点之前的节点）或者following-sibling（当前节点之后的节点）
        # xpath: //div[@class="page"]/a[@class="page_a"]/following-sibling::a[1]
        # 可以获取 class= "page_a"的a标签后相邻的a标签

        # 因为这里使用了 selenium 所以应该获取 下一页的按钮
        next_page_button = chrome.find_element(by=By.CSS_SELECTOR, value='.page_a + a')

        if next_page_button.text == '尾页':
            # 已经没有下一面了
            break # 跳出循环
        else:
            page_num += 1

        time.sleep(3)
        next_page_button.click()

    # print('标题：', new_titles)
    # print('CNNVD编号: ', new_nos)
    save_to_xls(new_titles,new_nos)
    print('========== "{}"爬取完毕 =========='.format(name))

    time.sleep(3)


# 3.根据漏洞编号爬取
# 根据编号查找一次只有一个标题
# 上面用 selenium 是为了反爬
def spider_by_no(no):
    url = 'http://www.cnnvd.org.cn/web/vulnerability/queryLds.tag'
    params = {
        'qcvCnnvdid': no
    }
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/106.0.0.0 Safari/537.36 Edg/106.0.1370.37'
    }
    response = requests.post(url=url, params=params, headers=headers)


    print('========== 正在爬取CNNVD编号为 "{}" 的标题和编号 =========='.format(no))
    # 获取标题
    title_list = [deal_title_by_no(response)]
    no_list = [no]

    save_to_xls(title_list, no_list)
    print('========== "{}"爬取完毕 =========='.format(no))


# 保存数据到 xls
def save_to_xls(title_list,no_list):
    """
    title_list:标题
    no_list:CNNVD编号
    """
    wb = xlrd.open_workbook('data.xls', formatting_info=True)
    xwb = copy(wb) # xwb可以写操作，wb不行
    sheet = xwb.get_sheet(0)
    rows = len(sheet.get_rows()) # 获取当前行数

    for i in range(len(title_list)):

        sheet.write(rows+1,0,title_list[i])
        sheet.write(rows+1,1,no_list[i])
        rows += 1

    # 保存同名文件，达到覆盖修改的目的，其中未被修改的内容保持不变
    xwb.save('data.xls')


# 处理标题（根据 CNNVD 编号爬取的）
# 有些标题过长，爬取的时候需要处理一下
# 返回一个字符串
def deal_title_by_no(response):
    response.encoding = 'utf-8'
    tree = etree.HTML(response.text)

    # 标题的xpath
    # 看看名字是不是过长的
    name_xpath = '//div[@class="list_list"]//li//div[@class="fl"]/a/attribute::onmouseover'
    name = tree.xpath(name_xpath)
    # 标题是正常的长度
    if len(name) == 0:
        name_xpath = '//div[@class="list_list"]//li//div[@class="fl"]/a/text()'
        title = str(tree.xpath(name_xpath)[0])
        # 需要再去掉特殊字符
        return title.strip()

    else:
        # 标题长度过长
        start = str(name).rfind(',')
        end = str(name).rfind("'")
        return str(name)[start + 2:end]



# 处理标题（使用 selenium 爬取的）
# 有些标题过长，爬取的时候需要处理一下
# 返回一个列表
# content: 页面源码
def deal_title_by_selenium(content):

    tree = etree.HTML(content)

    name_xpath = '//div[@class="list_list"]//li//div[@class="fl"]/a/text()'
    name_length = len(tree.xpath(name_xpath)) # 记录当前页有多少标题
    title_list = []

    # 标题的xpath
    # 看看名字是不是过长的
    # xpath = '//div[@class="list_list"]//li//div[@class="fl"]/a/attribute::onmouseover'

    for i in range(name_length):
        name = str(tree.xpath(name_xpath)[i]).strip()
        print(name)
        # 标题是正常的长度
        if name.find('...') != -1:
            # 标题过长
            a_xpath = '//div[@class="list_list"]//li//div[@class="fl"]/a'
            # 获取该结点对应属性
            title = tree.xpath(a_xpath)[i].attrib['onmouseover']

            # 标题长度过长
            start = str(title).rfind(',')
            end = str(title).rfind("'")
            title_list.append(str(title)[start + 2:end])


        else:
            # 标题正常
            title_list.append(name.strip())


    return title_list


if __name__ == '__main__':
    name_values,no_values = read_data('工作簿1.xlsx')

    for name in name_values:
        spider_by_selenium(name)
    # 之后爬取不用 selenium
    chrome.close()
    print("========== 10秒后开始根据 CNNVD 编号开始爬取 ==========")
    time.sleep(10)
    for no in no_values:
        spider_by_no(no)
        time.sleep(5)
