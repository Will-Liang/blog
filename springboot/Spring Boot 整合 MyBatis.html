    <!--导入springboot父工程-规定写法-->

        <artifactId>spring-boot-starter-parent</artifactId>
        <groupId>org.springframework.boot</groupId>
        <version>2.5.3</version>
    </parent>

    <!--引入相关的依赖-->
    <dependencies>
        <!--引入web starter-->

        <dependency>
            <groupId>org.mybatis.spring.boot</groupId>
            <artifactId>mybatis-spring-boot-starter</artifactId>

        </dependency>

        <!--引入mysql驱动: 这使用版本仲裁 8.0.26-->
        <dependency>
            <groupId>mysql</groupId>
            <artifactId>mysql-connector-java</artifactId>
        </dependency>

        <!--引入配置处理器 -->
        <dependency>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-configuration-processor</artifactId>
        </dependency>

        <!--引入lombok-->
        <dependency>
            <groupId>org.projectlombok</groupId>
            <artifactId>lombok</artifactId>
        </dependency>

        <!--引入test starter-->
        <dependency>
            <groupId>org.springframework.boot</groupId>

        <dependency>

    </dependencies>

</project>

    # 创建 src/main/java/com/lzw/springboot/mybatis/config/DruidDataSourceConfig.java

    package com.lzw.springboot.mybatis.config;

import com.alibaba.druid.pool.DruidDataSource;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import javax.sql.DataSource;
import java.sql.SQLException;

/**
 * @author LiAng
 */
@Configuration
public class DruidDataSourceConfig {
    @ConfigurationProperties("spring.datasource")
    @Bean
    public DataSource dataSource() throws SQLException {
        DruidDataSource druidDataSource = new DruidDataSource();
        return druidDataSource;
    }
}

    # 创建 src/main/java/com/lzw/springboot/mybatis/mapper/MonsterMapper.java

    package com.lzw.springboot.mybatis.mapper;

import com.lzw.springboot.mybatis.bean.Monster;
import org.apache.ibatis.annotations.Mapper;

/**
 * @author LiAng
 * 在Mapper接口使用 @Mapper 就会扫描，并将Mapper接口对象注入
 */
@Mapper
public interface MonsterMapper {
    //根据id返回Monster对象
    public Monster getMonsterById(Integer id);
}

    # 创建 src/main/resources/mapper/MonsterMapper.xml

    <?xml version="1.0" encoding="UTF-8"?>

        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.lzw.springboot.mybatis.mapper.MonsterMapper">

</mapper>

  #通过config-location 可以指定mybatis-config.xml，可以用传统的方式来配置mybatis
  #config-location: classpath:mybatis-config.xml

  #我们也可以直接在application.yml进行配置
  #举例说明1. 比如配置原来的 typeAliases
  #举例说明2 配置输出底层的原生sql
  type-aliases-package: com.lzw.springboot.mybatis.bean
  configuration:

    # 配置mybatis的方式二

    创建 src/main/resources/mybatis-config.xml

    <?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
        PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>

    <!--配置MyBatis自带的日志输出-查看原生的sql-->
    <settings>
        <setting name="logImpl" value="STDOUT_LOGGING"/>
    </settings>

    <!--
      1. 如果一个包下有很多的类，我们可以直接引入包
      2. 这样该包下面的所有类名，可以直接使用
    -->
    <typeAliases>
        <package name="com.lzw.springboot.mybatis.bean"/>
    </typeAliases>
</configuration>

    配置mybatis的两种方式的选择：如果配置比较简单，就直接在application.yml配置即可。如果配置内容比较多，可以考虑单独的做一个mybatis-config.xml。

    # 创建 src/main/java/com/lzw/springboot/mybatis/service/MonsterService.java

import com.lzw.springboot.mybatis.bean.Monster;

/**
 * @author LiAng
 */
public interface MonsterService {
    //根据id返回Monster对象

    # 创建 src/main/java/com/lzw/springboot/mybatis/service/impl/MonsterServiceImpl.java

    package com.lzw.springboot.mybatis.service.impl;

import com.lzw.springboot.mybatis.bean.Monster;
import com.lzw.springboot.mybatis.mapper.MonsterMapper;
import com.lzw.springboot.mybatis.service.MonsterService;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;

/**
 * @author LiAng
 */
@Service
public class MonsterServiceImpl implements MonsterService {

    //装配 MonsterMapper
    @Resource
    private MonsterMapper monsterMapper;

    @Override
    public Monster getMonsterById(Integer id) {
        return monsterMapper.getMonsterById(id);
    }
}

    # 创建 src/main/java/com/lzw/springboot/mybatis/controller/MonsterController.java

    package com.lzw.springboot.mybatis.controller;

import com.lzw.springboot.mybatis.bean.Monster;
import com.lzw.springboot.mybatis.service.MonsterService;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.ResponseBody;

import javax.annotation.Resource;

/**
 * @author LiAng
 */
@Controller
public class MonsterController {

    @Resource
    private MonsterService monsterService;

    @ResponseBody
    @GetMapping("/monster")
    public Monster getMonsterById(@RequestParam(value = "id") Integer id){
        return monsterService.getMonsterById(id);
    }
}
